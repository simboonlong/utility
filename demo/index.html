ease

Easing on any numerical attribute, of any element.

ease({
  startValue: 0,
  endValue: 1,
  decimal: 2,
  onStep: value => {
    document.getElementById("ease-opacity").style.opacity = value
  }
})  
ease({
  startValue: 0,
  endValue: 800,
  onStep: value => {
    document.getElementById("ease-scroll").scrollLeft = value
  }
})

scrollProgressBody, scrollProgressItem

Indication of scroll progress.

const progress = scrollProgressBody({
  scrollTopCurr
});

document.getElementById("sp-body").innerText = `${progress}%`;

const progress = scrollProgressItem({
  element: document.getElementById("sp-item"),
  scrollTopCurr
});

document.querySelector(".progress-thumb").style.width = `${(progress * 100).toFixed(0)}%`;
